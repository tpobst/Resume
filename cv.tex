\section{Research Projects}
\cventry{2010-2013}{Radiation Detector Characterization}{}{Knoxville, TN}{}{
  \emph{Developed characterization techniques allowing for the measurement of thin films for portal monitor applications}
\begin{itemize}
	\item Advanced repeatable measurement techniques
  \item Developed techniques for gamma discrimination based on pulse height discriminators
	\item Characterized detector light yield,neutronic performance, and gamma discrimination utilizing an authored MATLAB code base
\end{itemize}}
\cventry{2012--2013}{Radiation Detector Simulation}{}{Knoxville, TN}{}{
  \emph{Simulations of various detector assemblies with MCNPX, SCALE and GEANT4 providing insights for effective detector design}
  \begin{itemize}
    \item Designed and fabricated a ${}^{60}$Co irradiator in compliance with radiation safety regulations
    \item Calculated intrinsic efficiencies based MCNPX calculations and measured spectra
    \item Simulated various neutron irradiator for calculation benchmarking and preformed parameter studies and optimization of a replacement radiation portal monitor
    \item Simulated light transport in various detector geometries and materials 
  \end{itemize}}
\cventry{2013}{Spent Fuel Cask NDA}{}{Knoxville, TN}{}{
  Designed and evaluated gamma collimators for the in-situ NDA of spent fuel casks
\begin{itemize}
	\item Completed MCNPX simulations utilizing variance reduction of various collimator design with point and area sources
	\item Validation of the collimator designs by generalized geometry holdup measurements
\end{itemize}}
\cventry{2012}{Pulse Shape Discrimination}{}{Knoxville, TN}{}{Preformed pulse shape discrimination using charge integration, modal methods, and artificial neural networks in the MATLAB environment }
